
Function: ErrorDetection(data)
1.  total_ones = 0  // Initialize the count of ones
2.  FOR each bit in data:
3.      total_ones += bit  // Count the number of ones
4.  parity_bit = (total_ones % 2 == 0) ? 0 : 1  // Calculate the parity bit (even parity)
5.  data_with_parity = data + parity_bit  // Append the parity bit to the data
6.  // Simulate an error (optional)
7.  received_data = data_with_parity  // Assuming data reception
8.  received_parity_bit = received_data[-1]  // Extract the received parity bit
9.  calculated_parity_bit = 0  // Initialize calculated parity bit
10. FOR each bit in received_data[:-1]:  // Exclude parity bit while calculating
11.     calculated_parity_bit += bit  // Calculate the sum of bits
12. calculated_parity_bit = (calculated_parity_bit % 2 == 0) ? 0 : 1  // Calculate the received parity bit
13. IF received_parity_bit == calculated_parity_bit:
14.     PRINT "No error"
15. ELSE:
16.     PRINT "Error detected"
17. END IF
END Function
